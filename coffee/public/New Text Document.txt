<script>
// ═══════════════════════════════════════════════
//  LOGOUT
// ═══════════════════════════════════════════════
function logout() {
  sessionStorage.clear();
  localStorage.removeItem('mach_token');
  window.location.href = 'index.html';
}

// ═══════════════════════════════════════════════
//  AUTH GUARD
// ═══════════════════════════════════════════════
(function() {
  const role = sessionStorage.getItem('mach_role');
  if (!role) { window.location.href = 'index.html'; return; }

  if (role === 'owner') {
    const token = localStorage.getItem('mach_token');
    if (!token) { window.location.href = 'index.html'; return; }
  }

  // Passed — hide auth gate and boot
  document.getElementById('auth-gate').style.display = 'none';
  init();
})();

// ═══════════════════════════════════════════════
//  STATE
// ═══════════════════════════════════════════════
let currentMode = 'today';
let currentDate = todayStr();

function todayStr() { return new Date().toISOString().slice(0, 10); }
function subtractDays(n) { const d = new Date(); d.setDate(d.getDate() - n); return d.toISOString().slice(0, 10); }

// ═══════════════════════════════════════════════
//  DATE PICKER
// ═══════════════════════════════════════════════
document.getElementById('date-picker').addEventListener('change', function() {
  currentDate = this.value;
  currentMode = 'custom';
  document.querySelectorAll('.chip').forEach(c => c.classList.remove('active'));
  fetchOrders();
});
document.getElementById('date-picker').value = todayStr();

// ═══════════════════════════════════════════════
//  CHIP SELECTION
// ═══════════════════════════════════════════════
function selectChip(el, days) {
  document.querySelectorAll('.chip').forEach(c => c.classList.remove('active'));
  el.classList.add('active');

  if (days === -1) {
    currentMode = 'all';
    currentDate = null;
    document.getElementById('date-picker').value = '';
  } else if (days === 7) {
    currentMode = 'week';
    currentDate = null;
    document.getElementById('date-picker').value = '';
  } else {
    currentMode = days === 0 ? 'today' : 'yesterday';
    currentDate = subtractDays(days);
    document.getElementById('date-picker').value = currentDate;
  }
  fetchOrders();
}

// ═══════════════════════════════════════════════
//  AVAILABLE DATE CHIPS
// ═══════════════════════════════════════════════
function renderAvailableDates(dates) {
  const container = document.getElementById('available-dates');
  if (!dates.length) { container.innerHTML = '<span style="color:rgba(255,255,255,0.3);font-size:0.7rem">None yet</span>'; return; }
  container.innerHTML = dates.slice(0, 10).map(d =>
    `<div class="chip" style="font-size:0.65rem;padding:0.2rem 0.5rem" onclick="jumpToDate('${d}')">${d}</div>`
  ).join('');
}

function jumpToDate(dateStr) {
  document.querySelectorAll('.chip').forEach(c => c.classList.remove('active'));
  currentDate = dateStr;
  currentMode = 'custom';
  document.getElementById('date-picker').value = dateStr;
  fetchOrders();
}

// ═══════════════════════════════════════════════
//  CARD CLASS
// ═══════════════════════════════════════════════
function cardClass(printers) {
  const has2 = printers.includes(2), has3 = printers.includes(3);
  if (has2 && has3) return 'p123';
  if (has2)         return 'p12';
  if (has3)         return 'p13';
  return 'p1only';
}

// ═══════════════════════════════════════════════
//  FETCH ORDERS
// ═══════════════════════════════════════════════
async function fetchOrders() {
  try {
    let url = '/api/orders/routing';
    if (currentMode === 'today' || currentMode === 'yesterday' || currentMode === 'custom') {
      url += `?date=${currentDate}`;
    }

    const token = localStorage.getItem('mach_token') || '';
    const res   = await fetch(url, {
      headers: { ...(token && { 'Authorization': 'Bearer ' + token }) }
    });
    const data = await res.json();
    if (!data.success) throw new Error('API error');

    let orders = data.orders;

    if (currentMode === 'week') {
      const cutoff = subtractDays(6);
      orders = orders.filter(o => o.timestamp.slice(0, 10) >= cutoff);
    }

    renderAvailableDates(data.availableDates || []);
    document.getElementById('last-updated').textContent = 'Updated: ' + new Date().toLocaleTimeString('en-IN');

    document.getElementById('stat-total').textContent   = orders.length;
    document.getElementById('stat-revenue').textContent = '₹' + orders.reduce((s, o) => s + o.grand, 0).toLocaleString('en-IN');
    document.getElementById('stat-p1').textContent      = orders.filter(o => o.printers.includes(1)).length;
    document.getElementById('stat-p2').textContent      = orders.filter(o => o.printers.includes(2)).length;
    document.getElementById('stat-p3').textContent      = orders.filter(o => o.printers.includes(3)).length;

    const container = document.getElementById('orders-container');
    if (!orders.length) {
      const label = currentMode === 'all'  ? 'No orders found.' :
                    currentMode === 'week' ? 'No orders in the last 7 days.' :
                    `No orders on ${currentDate}.`;
      container.innerHTML = `<div class="empty">📭 ${label}</div>`;
      return;
    }

    container.innerHTML = orders.map(o => `
      <div class="order-card ${cardClass(o.printers)}">
        <div>
          <div class="order-top">
            <span class="inv-no">${o.invoiceNo}</span>
            <span class="table-tag">${o.tableNo}</span>
            <span class="status-tag status-${o.status}">${o.status}</span>
            <span class="time-tag">${new Date(o.timestamp).toLocaleString('en-IN', { dateStyle: 'medium', timeStyle: 'short' })}</span>
          </div>
          <div class="printer-badges">
            ${o.printers.map(p => `
              <span class="p-badge p${p}">🖨 Printer ${p} — ${p === 1 ? 'Full Invoice' : p === 2 ? 'Beverages' : 'Food Items'}</span>
            `).join('')}
          </div>
          <div class="items-grid">
            ${o.breakdown.beverages.length ? `
              <div class="items-col">
                <h4>☕ Beverages → Printer 2</h4>
                ${o.breakdown.beverages.map(i => `<div class="item-row"><span>${i.name}</span><span>x${i.qty} · ₹${i.amount}</span></div>`).join('')}
              </div>` : ''}
            ${o.breakdown.food.length ? `
              <div class="items-col">
                <h4>🍛 Food → Printer 3</h4>
                ${o.breakdown.food.map(i => `<div class="item-row"><span>${i.name}</span><span>x${i.qty} · ₹${i.amount}</span></div>`).join('')}
              </div>` : ''}
            ${!o.breakdown.beverages.length && !o.breakdown.food.length ? `
              <div class="items-col">
                <h4>📄 All Items → Printer 1 Only</h4>
                ${o.breakdown.all.map(i => `<div class="item-row"><span>${i.name}</span><span>x${i.qty} · ₹${i.amount}</span></div>`).join('')}
              </div>` : ''}
          </div>
        </div>
        <div class="order-right">
          <div class="grand-label">Grand Total</div>
          <div class="grand-amount">₹${o.grand.toLocaleString('en-IN')}</div>
          <div class="customer-name">👤 ${o.customer}</div>
          <div class="customer-phone">📞 ${o.phone}</div>
        </div>
      </div>
    `).join('');

  } catch (e) {
    document.getElementById('orders-container').innerHTML =
      `<div class="error">⚠️ Could not connect to server.<br>
        <small style="color:#8A6B55;font-size:0.8rem;margin-top:0.5rem;display:block">
          Make sure <strong>node server.js</strong> is running and you opened this via
          <strong>http://localhost:3000/printer-dashboard.html</strong>
        </small>
      </div>`;
  }
}

// ═══════════════════════════════════════════════
//  INIT
// ═══════════════════════════════════════════════
function init() {
  fetchOrders();
  setInterval(fetchOrders, 5000);
}
